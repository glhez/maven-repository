64     * Where are file to be copied found?
65     * <p>
66     * The plugin provide two way of locating file:
67     * <ul>
68     * <li>From dependencies, using {@code jar:}.</li>

77    /**
78     * The maven-eclipse-plugin allows you to move settings files from one
79     * location to another. You use that to put each configuration file from
80     * your settings JAR in the right location:
81     *
82     * <pre>
83     * &lt;plugin&gt;

85     *     &lt;configuration&gt;

91     *             &lt;file&gt;

95     *                 &lt;failOnError&gt;false&lt;failOnError&gt;
96     *             &lt;/file&gt;

98     *         &lt;/additionalConfig&gt;
99     *     &lt;/configuration&gt;
100    * &lt;/plugin&gt;

119   @Override
120   public void execute() throws MojoExecutionException, MojoFailureException {
121     try {
122       if (skip) {
123         getLog().info("Skipping project settings configuration.");
124         return;
125       }
126       if (validateMojo()) {
127         try (ResourceResolver resolver = getResourceResolver()) {
128           writeAdditionalConfig(resolver, filterAdditionalConfig());
129         }
130         getLog().info("Project configured");
131       } else {
132         throw new MojoFailureException("project not configured due to errors");
133       }
134     } catch (final IOException e) {
135       getLog().error("Failure during settings configuration", e);
136     }
137   }

139   private ResourceResolver getResourceResolver() throws IOException, MojoExecutionException {
140     ResourceResolver resolver;
141     String prefix;
142     if (source.equals("jar") || source.startsWith("jar:")) {
143       prefix = source.equals("jar") ? "" : source.substring("jar:".length());
144       resolver = new ArtifactResourceResolver(new ArtifactCollector(project, getLog(), plugin).collect(), prefix);
145     } else if (source.equals("file") || source.startsWith("file:")) {
146       prefix = source.equals("file") ? "" : source.substring("file:".length());
147       resolver = new FileSystemResourceResolver(
148           prefix.isEmpty() ? project.getBasedir() : new File(prefix).getAbsoluteFile());
149     } else {
150       throw new MojoExecutionException("<source> does not start with 'jar:' or 'file:' ");
151     }
152     if (getLog().isDebugEnabled()) {
153       getLog().debug("Resolving file using " + resolver);
154     }
155     return resolver;
156   }

158   private boolean validateMojo() throws MojoExecutionException {
159     this.source = StringUtils.trimToNull(source);
160     if (source == null) {
161       throw new MojoExecutionException("<source> is missing.");
162     }
163 
164     if (additionalConfig == null || additionalConfig.length < 1) {
165       addError("No <additionalConfig> provided");
166       return false;
167     }
168 
169     boolean fail = false;
170     int index = 0;
171     for (EclipseSettingsFile config : additionalConfig) {
172       config.setLocation(StringUtils.trimToNull(config.getLocation()));
173       config.setName(StringUtils.trimToNull(config.getName()));
174       if (config.getFailOnError() == null) {
175         config.setFailOnError(failOnError);
176       }
177 
178       if (config.getLocation() == null) {
179         addError(
180                  "Invalid <location> in AdditionalConfig[%d]%s.",
181                    index,
182                    formatInputLocation(config.getInputLocation()));
183         fail = true;
184       }
185       if (config.getName() == null) {
186         addError(
187                  "Invalid <name> in AdditionalConfig[%d]%s.",
188                    index,
189                    formatInputLocation(config.getInputLocation()));
190         fail = true;
191       }
192       ++index;
193     }
194 
195     return !fail;
196   }

198   private void addError(String format, Object... params) {
199     addMessage(format, BuildContext.SEVERITY_ERROR, params);
200   }
201 
202   private void addMessage(String format, int severity, Object... params) {
203     String message;
204     Throwable throwable = null;
205     if (params.length == 0) {
206       message = format;
207     } else {
208       int n = params.length - 1;
209       if (n >= 0 && params[n] instanceof Throwable) {
210         message = String.format(format, Arrays.copyOf(params, n));
211         throwable = (Throwable) params[n];
212       } else {
213         message = String.format(format, params);
214       }
215     }
216     buildContext.addMessage(project.getFile(), 0, 0, message, severity, throwable);
217   }

219   private EclipseSettingsFile[] filterAdditionalConfig() {
220     final PackagingFilter packagingFilter = PackagingFilter.newPackagingFilter(packagings);
221     final String projectPackaging = project.getPackaging();
222 
223     List<EclipseSettingsFile> files = new ArrayList<>(additionalConfig.length);
224     for (final EclipseSettingsFile config : additionalConfig) {
225       final PackagingFilter localPackagingFilter = packagingFilter.join(config.getPackagings());
226       if (!localPackagingFilter.test(projectPackaging)) {
227         if (getLog().isDebugEnabled()) {
228           getLog().debug(
229                          "Ignoring " + config.getLocation() + " -> " + config.getName() + " because packaging '"
230                              + projectPackaging + "' is filtered by " + localPackagingFilter);
231         }
232         continue;
233       }
234       files.add(config);
235     }
236 
237     if (files.size() != additionalConfig.length) {
238       getLog().info("Filtered " + (additionalConfig.length - files.size()) + " files due to packaging.");
239       return files.toArray(new EclipseSettingsFile[0]);
240     }
241     return additionalConfig;
242   }

244   private void writeAdditionalConfig(final ResourceResolver resolver, final EclipseSettingsFile[] additionalConfig)
245       throws MojoExecutionException {
246     getLog().info("Will copy " + additionalConfig.length + " using " + resolver);
247 
248     final List<IOException> failures = new ArrayList<>();
249     final File basedir = project.getBasedir();
250 
251     List<File> updatedFiles = new ArrayList<>();
252     for (final EclipseSettingsFile config : additionalConfig) {
253       final String location = config.getLocation();
254       final File target = new File(basedir, config.getName());
255       try {
256         writeAdditionalConfig(resolver, config.getLocation(), target);
257         updatedFiles.add(target);
258       } catch (final IOException e) {
259         addMessage(config, copyFailure(config, location, target, e), e);
260         if (config.isFailOnError()) {
261           failures.add(e);
262         }
263       }
264     }
265 
266     if (!failures.isEmpty()) {
267       final MojoExecutionException ex = new MojoExecutionException("Unable to copy " + failures.size() + " files");
268       for (final IOException e : failures) {
269         ex.addSuppressed(e);
270       }
271       throw ex;
272     }
273     for (File updatedFile : updatedFiles) {
274       buildContext.refresh(updatedFile);
275     }
276   }
277 
278   private void addMessage(EclipseSettingsFile config, String message, IOException e) {
279     int severity = config.isFailOnError() ? BuildContext.SEVERITY_ERROR : BuildContext.SEVERITY_WARNING;
280     buildContext.addMessage(project.getFile(), 0, 0, message, severity, e);
281   }

283   private String copyFailure(EclipseSettingsFile config, String location, File target, IOException e) {
284     InputLocation inputLocation = config.getInputLocation();
285     return "Could not copy <" + location + "> to <" + target.getAbsolutePath() + ">"
286         + formatInputLocation(inputLocation) + ": " + e.getMessage();
287   }
288 
289   private void writeAdditionalConfig(ResourceResolver resolver, String location, File target) throws IOException {
290     final Resource source = resolver.getResource(location);
291     if (source == null) {
292       throw new FileNotFoundException("Missing resource: " + location + " using " + resolver);
293     }
294 
295     /*
296      * if we try to write to a directory, this will fail. We don't try to behave like mv which
297      * relocate the source into the folder (eg: mv a b will give b/a).
298      */
299     if (target.isDirectory()) {
300       getLog().warn(target.getAbsolutePath() + " is a directory, ignoring.");
301       return;
302     }
303     target.getParentFile().mkdirs();
304 
305     getLog().info("Copying " + source + " to " + target);
306     try (InputStream inStream = source.getResourceAsStream();
307         OutputStream outStream = new FileOutputStream(target)) {
308       IOUtil.copy(inStream, outStream);
309     }
